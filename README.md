### Prime Numbers

```python

import math

# Simple approach
def isPrime1(n):
    if n < 2:
        return False
    for i in range(2, n):
        if n % i == 0:
            return False
    return True

# Slightly optimized
def isPrime2(n):
    if n < 2:
        return False
    for i in range(2, int(math.sqrt(n))+1):
        if n % i == 0:
            return False
    return True 

# More optimized
def isPrime3(n):
    if n < 2:
        return False
    if n % 2 == 0 and n > 2:
        return False
    for i in range(3, int(math.sqrt(n))+1, 2):
        if n % i == 0:
            return False
    return True 

N = int(input())
print(isPrime3(N))

```

def convertToBase2(n):
    s = ''
    while n > 0:
        if n % 2 == 0:
            s = '0' + s
        else:
            s = '1' + s
        n >>= 1
    return s

def convertBinaryToBase10(s):
    n = 0
    d = 1
    for i in reversed(range(len(s))):
        n += int(s[i]) * d
        d <<= 1
    return n

N = int(input('Enter base10 integer, i.e. 13: '))
print(convertToBase2(N))
s = input('Enter binary string, i.e. 1010: ')
print(convertBinaryToBase10(s))
